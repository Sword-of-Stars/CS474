\subsection*{Handling State $1$}

\begin{enumerate}
    \item \textcolor{gray}{Calculate the $\epsilon$-closure of the state in M; call the result $S$.} 
    From the $\epsilon$-closure table, we know that $E(\{ 1 \})=\{ {1, 3} \}$. 
    
    \item \textcolor{gray}{Calculate the transitions in $M$ from each state in $R$ on each input character.} 
    \begin{itemize}
        
            \item [\textcolor{red}{\textbf{ a }}] 
            Here we must collect all places we can get to when reading an \textcolor{red}{\textbf{ a }}
            starting in any state in $\{ {1, 3} \}$. 
            Since
            \begin{itemize}\item $\delta(1, \textcolor{red}{\textbf{ a }}) = \{2\}
$\item $\delta(3, \textcolor{red}{\textbf{ a }}) = \{5\}
$
            \end{itemize}
            we see then that $\delta_R(1,\textcolor{red}{\textbf{ a }})$ includes at least $\{ {2, 5} \}$.
        
            \item [\textcolor{red}{\textbf{ b }}] 
            Here we must collect all places we can get to when reading an \textcolor{red}{\textbf{ b }}
            starting in any state in $\{ {1, 3} \}$. 
            Since
            \begin{itemize}\item $\delta(1, \textcolor{red}{\textbf{ b }}) = \varnothing
$\item $\delta(3, \textcolor{red}{\textbf{ b }}) = \{1,2,5\}
$
            \end{itemize}
            we see then that $\delta_R(1,\textcolor{red}{\textbf{ b }})$ includes at least $\{ {1, 2, 5} \}$.
        
    \end{itemize}

    \item \textcolor{gray}{For each input character, calculate the $\epsilon$-closure of the resulting set of states.}   
    \begin{itemize}
    
            \item [\textcolor{red}{\textbf{ a }}] 
            Merging the relevant rows in the table for $E(\{ {2, 5} \})$, 
            we see that $\delta_R( 1, a )=\{ {2, 5} \}$. 
        
            \item [\textcolor{red}{\textbf{ b }}] 
            Merging the relevant rows in the table for $E(\{ {1, 2, 5} \})$, 
            we see that $\delta_R( 1, b )=\{ {1, 2, 3, 5} \}$. 
        
    \end{itemize}
\end{enumerate}
