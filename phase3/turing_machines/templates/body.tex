{%- macro format_state(state) -%}
{%- if state == 'qaccept' -%}
q_{\text{accept}}
{%- elif state == 'qreject' -%}
q_{\text{reject}}
{%- elif state and state|string|length > 1 and state[0] == 'q' -%}
q_{ {{- state[1:] -}} }
{%- else -%}
{{ state }}
{%- endif -%}
{%- endmacro -%}

{%- macro header_sym(s) -%}
{%- if s == "#" -%}\#{%-
 elif s == "\\#" or s == "\\\\#" -%}\#{%-
 elif s == "\\sqcup" or s == "\\\\sqcup" -%}\sqcup{%-
 else -%}{{ s }}{%- endif -%}
{%- endmacro -%}
{%- macro format_symbol(sym, blank) -%}
{%- if sym == '#' -%}
\Sym{\#}
{%- elif sym == blank -%}
\Sym{\ensuremath{\sqcup}}
{%- elif sym == '0' -%}
\Sym{0}
{%- elif sym == '1' -%}
\Sym{1}
{%- elif sym == 'x' -%}
\Sym{x}
{%- else -%}
\Sym{ {{- sym -}} }
{%- endif -%}
{%- endmacro -%}
